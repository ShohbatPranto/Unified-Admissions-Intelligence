--Cleaning Datasets
-- 1. Clean APPLICANT Table
DROP TABLE IF EXISTS applicant_clean;

CREATE TABLE applicant_clean AS
SELECT
    TRIM(Reference_ID) AS Reference_ID,
    Given_Name,
    Last_Name,
    Major,
    Country,
    City, -- <--- Added City Here
    -- Fix: Treat text "NULL" as empty before converting math
    TO_TIMESTAMP(NULLIF(NULLIF(Recieved_At, ''), 'NULL')::NUMERIC / 1000) AS Recieved_At,
    University,
    Citizenship,
    Intake,
    College_1st_Choice,
    Email_ID,
    TO_DATE(NULLIF(NULLIF(Date_of_Birth, ''), 'NULL'), 'MM/DD/YYYY') AS Date_of_Birth,
    Application_Source,
    TRIM(SEVIS_ID) AS SEVIS_ID,
    TO_TIMESTAMP(NULLIF(NULLIF(Created_At, ''), 'NULL')::NUMERIC / 1000) AS Created_At,
    Most_Recent_Released_Decision,
    TO_DATE(NULLIF(NULLIF(Admit_Date, ''), 'NULL'), 'MM/DD/YYYY') AS Admit_Date
FROM applicant_raw;
-- 2. Clean CONNECT Table
DROP TABLE IF EXISTS connect_clean;
CREATE TABLE connect_clean AS
SELECT
    TRIM(Reference_ID) AS Reference_ID,
    Deposit_Status,
    Assigned,
    -- Fix: Treat text "NULL" as empty before converting timestamp
    TO_TIMESTAMP(NULLIF(NULLIF(Recieved_At, ''), 'NULL'), 'MM/DD/YYYY HH24:MI') AS Recieved_At,
    University,
    I_20_Status,
    TO_TIMESTAMP(NULLIF(NULLIF(Created_At, ''), 'NULL')::NUMERIC / 1000) AS Created_At,
    TO_TIMESTAMP(NULLIF(NULLIF(Modified_At, ''), 'NULL')::NUMERIC / 1000) AS Modified_At
FROM connect_raw;

-- 3. Clean OUTREACH Table 
DROP TABLE IF EXISTS outreach_clean;
CREATE TABLE outreach_clean AS
SELECT
    Student_Full_Name,
    Program_Of_interest,
    Term,
    Application_Source,
    Deposit_Paid,
    TRIM(SEVIS_ID) AS SEVIS_ID,
    Person_Email,
    Citizenship_Primary,
    Student_Status,
    Remarks
FROM outreach_raw;

-- 4. Clean SEVIS Table
DROP TABLE IF EXISTS sevis_clean;
CREATE TABLE sevis_clean AS
SELECT
    TRIM(SEVIS_ID) AS SEVIS_ID,
    SEVIS_Status,
    TO_DATE(NULLIF(NULLIF(Status_Change_Date, ''), 'NULL'), 'MM/DD/YYYY') AS Status_Change_Date,
    Surname_Or_Primary_Name,
    Given_Name,
    Country_of_Citizenship,
    Gender,
    Email,
    TO_DATE(NULLIF(NULLIF(Visa_Expiration_Date, ''), 'NULL'), 'MM/DD/YYYY') AS Visa_Expiration_Date,
    Education_Level,
    Major_1_Description,
    TO_DATE(NULLIF(NULLIF(Program_Start_Date, ''), 'NULL'), 'MM/DD/YYYY') AS Program_Start_Date,
    TO_DATE(NULLIF(NULLIF(Program_End_Date, ''), 'NULL'), 'MM/DD/YYYY') AS Program_End_Date,
    -- NEW FINANCIAL COLUMNS
    -- We cast these to NUMERIC so you can sum them up in the dashboard
    CAST(NULLIF(Tuition_Fees, '') AS NUMERIC) AS Tuition_Fees,
    CAST(NULLIF(Living_Expenses, '') AS NUMERIC) AS Living_Expenses,
    CAST(NULLIF(Dependent_Expenses, '') AS NUMERIC) AS Dependent_Expenses,
    CAST(NULLIF(Other_Costs, '') AS NUMERIC) AS Other_Costs,
    CAST(NULLIF(Students_Personal_Funds, '') AS NUMERIC) AS Students_Personal_Funds,
    CAST(NULLIF(Funds_From_This_School, '') AS NUMERIC) AS Funds_From_This_School, -- Scholarships usually
    CAST(NULLIF(Funds_From_Other_Sources, '') AS NUMERIC) AS Funds_From_Other_Sources,
    CAST(NULLIF(On_Campus_Employment, '') AS NUMERIC) AS On_Campus_Employment
FROM sevis_raw;

--Joining the datasets
DROP TABLE IF EXISTS master_report;


--Final Master Join
DROP TABLE IF EXISTS master_report_final;

CREATE TABLE master_report_final AS
SELECT DISTINCT ON (a.Reference_ID)
    -- Applicant Info
    a.Reference_ID,
    a.Given_Name,
    a.Last_Name,
    a.Major,
    a.Country,
    a.City, -- <--- Added City Here
    
    -- FORCE DATES TO BE SIMPLE DATES (Fixes Looker Studio Issue)
    CAST(a.Recieved_At AS DATE) AS Application_Date, 
    
    a.University,
    a.Citizenship,
    a.Intake,
    a.College_1st_Choice,
    a.Email_ID,
    a.Date_of_Birth,
    a.Application_Source,
    a.SEVIS_ID,
    a.Most_Recent_Released_Decision,
    a.Admit_Date,

    -- Connect Info
    c.Deposit_Status AS Connect_Deposit_Status,
    c.Assigned AS Connect_Assigned_To,
    CAST(c.Recieved_At AS DATE) AS Connect_Recieved_Date, -- Fixed Date
    c.I_20_Status AS Connect_I20_Status,

    -- Outreach Info
    o.Student_Status AS Outreach_Student_Status,
    o.Full_Remarks_History AS Outreach_Remarks,
    o.Deposit_Paid AS Outreach_Deposit_Paid,
    o.Program_Of_interest,
    o.Term,

    -- SEVIS Info
    s.SEVIS_Status,
    s.Visa_Expiration_Date,
    s.Program_Start_Date,
    s.Program_End_Date,
    s.Education_Level,
    s.Major_1_Description AS SEVIS_Major,

    -- Financials
    s.Tuition_Fees,
    s.Living_Expenses,
    s.Dependent_Expenses,
    s.Other_Costs,
    s.Students_Personal_Funds,
    s.Funds_From_This_School,
    s.Funds_From_Other_Sources,
    s.On_Campus_Employment,
    
    -- Calculated Fields
    (COALESCE(s.Tuition_Fees, 0) + COALESCE(s.Living_Expenses, 0) + COALESCE(s.Dependent_Expenses, 0) + COALESCE(s.Other_Costs, 0)) AS Total_Estimated_Cost,
    
    ((COALESCE(s.Tuition_Fees, 0) + COALESCE(s.Living_Expenses, 0) + COALESCE(s.Dependent_Expenses, 0) + COALESCE(s.Other_Costs, 0)) - 
     (COALESCE(s.Students_Personal_Funds, 0) + COALESCE(s.Funds_From_This_School, 0) + COALESCE(s.Funds_From_Other_Sources, 0) + COALESCE(s.On_Campus_Employment, 0))) AS Funding_Gap

FROM applicant_clean a
LEFT JOIN connect_latest c ON a.Reference_ID = c.Reference_ID
LEFT JOIN outreach_latest o ON a.SEVIS_ID = o.SEVIS_ID
LEFT JOIN sevis_clean s ON a.SEVIS_ID = s.SEVIS_ID
ORDER BY a.Reference_ID;
